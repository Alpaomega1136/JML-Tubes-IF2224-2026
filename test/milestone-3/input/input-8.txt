{pengujian 8 - constraint array dan record}

program KesalahanArrayRecord;
type
    Titik == record
        x : integer;
        y : integer;
    end;
    Daftar == array [1..3] of integer;
var
    p      : Titik;
    data   : Daftar;
    indeks : string;
    angka  : integer;
begin
    indeks := 'dua';

    data[indeks] := 10;
    data[5] := 20;
    p.z := 3;
    angka.x := 1;
end.
