    MAX := 10;
    PI  := 3.14159;

    Titik = record
        x : real;
        y : real;
    end;
    Daftar = array [1..10] of integer;
var
    p    : Titik;
    data : Daftar;
    i    : integer;

function Kuadrat(n : integer) : integer;
begin
    Kuadrat := n * n;
end;

procedure IsiArray(var arr : Daftar; n : integer);
var
    j : integer;
begin
    for j := 1 to n do
        arr[j] := j * j;
end;
